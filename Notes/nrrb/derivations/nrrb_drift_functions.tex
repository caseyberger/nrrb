\section{\label{NRRBCLM}Generating the NRRB CL Equations}

\subsection{\label{LatticeDerivatives} Derivatives on the Lattice}
When taking derivatives of this lattice action with respect to the fields, we do the following:

%
\bea
\frac{\delta}{\delta \phi_{c,r}}\left(\sum_{q=1}^{N_{r}}\sum_{a=1}^{2}\phi_{a,q}\phi_{a,q+\hat{i}}\right)& =& \sum_{a=1}^{2}\sum_{q=1}^{N_{r}}\left( \phi_{a,q} \frac{\delta}{\delta  \phi_{c,r}} \phi_{a,q+\hat{i}} +\frac{\delta  \phi_{a,q}}{\delta  \phi_{c,r}}  \phi_{a,q+\hat{i}} \right)\\
&= & \sum_{a=1}^{2}\sum_{q=1}^{N_{r}}\left( \phi_{a,q}\delta_{c,a} \delta_{r,q+\hat{i}} +\delta_{c,a} \delta_{q,r} \phi_{a,q+\hat{i}} \right) \nonumber\\
&= &  \phi_{c,r-\hat{i}} +\phi_{c,r+\hat{i}} . \nonumber
\eea
%
Similarly, 
%
\beq
\frac{\delta}{\delta \phi_{c,r}}\left(\sum_{q=1}^{N_{r}}\sum_{a=1}^{2}\sum_{b=1}^{2}\epsilon_{ab}\phi_{a,q}\phi_{b,q+\hat{i}}\right) = \sum_{b=1}^{2}\epsilon_{cb}\left( \phi_{b,r-\hat{i}} +\phi_{b,r+\hat{i}} \right).
\eeq
%

\subsection{\label{KAlgebra}Computing the derivative of the action with respect to the real fields}
The first step in computing the CL Equations is to find $\frac{\delta S_{r}}{\delta \phi_{a,r}}$. This is done below, with the sum over $a = 1,2$ implied:
%
\bea
\frac{\delta S_{r}}{\delta \phi_{a,r}}  & =& \frac{\delta S_{\mu,r}}{\delta \phi_{a,r}} +  \frac{\delta S_{\del,r}}{\delta \phi_{a,r}} -\frac{\delta S_{\text{trap},r} }{\delta \phi_{a,r}} - \frac{\delta S_{\omega,r}}{\delta \phi_{a,r}}  + \frac{\delta S_{\text{int},r}}{\delta \phi_{a,r}} \\
&& \nonumber
\eea
%
Again, we proceed by modifying each of the 5 parts of the action. First, the time and chemical potential term:
%
\bea
\frac{\delta}{\delta \phi_{a,r}} S_{\mu,r} & =& \frac{1}{2} \frac{\delta}{\delta \phi_{a,r}}\sum_{c=1}^{2}\left[\phi_{c,r}^{2} - e^{ \bar{\mu}}\phi_{c,r}\phi_{c,r - \hat{\tau}} - i e^{ \bar{\mu}} \sum_{b=1}^{2}\epsilon_{cb}  \phi_{c,r}\phi_{b,r - \hat{\tau}} \right]\nonumber\\
& =& \phi_{a,r}- \frac{1}{2} e^{\bar{\mu}}\left(\phi_{a,r - \hat{\tau}}+\phi_{a,r + \hat{\tau}}\right) - \frac{i}{2} e^{\bar{\mu}}\epsilon_{ab}\left(\phi_{b,r - \hat{\tau}} +\phi_{b,r + \hat{\tau}}\right)
\eea
%
Next, the spatial derivative part:
%
\bea
\frac{\delta}{\delta \phi_{a,r}} S_{\del,r} & =& \frac{1}{4 \bar{m}}\frac{\delta}{\delta \phi_{a,r}}\sum_{j=1}^{d}\sum_{c=1}^{2}\left[ 2\phi_{c,r}^{2} - (\phi_{c,r}\phi_{c,r + \hat{j}}+\phi_{c,r}\phi_{c,r - \hat{j}}) \right. \nonumber \\
&& \left. -i\sum_{b=1}^{2}\epsilon_{cb} (\phi_{c,r}\phi_{b,r + \hat{j}}+\phi_{c,r}\phi_{b,r - \hat{j}}) \right] \nonumber\\
& =& \frac{1}{4 \bar{m}}\sum_{j=1}^{d}\left[ 4\phi_{a,r} - 2(\phi_{a,r + \hat{j}}+\phi_{c,r - \hat{j}}) - 2 i\sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r + \hat{j}}+\phi_{b,r - \hat{j}}) \right] \nonumber\\
& =& \frac{1}{2 \bar{m}}\sum_{j=1}^{d}\left[ 2\phi_{a,r} - (\phi_{a,r + \hat{j}}+\phi_{c,r - \hat{j}}) -  i\sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r + \hat{j}}+\phi_{b,r - \hat{j}}) \right] 
\eea
%
%the imaginary part cancels out when taking the derivative - do this with a = 1,2 and b=1,2 written out explicitly to see it
Then the part of the action due to the external trapping potential:
%
\bea
\frac{\delta}{\delta \phi_{a,r}}  S_{\text{trap},r}& =&\frac{\bar{m}\bar{\omega}_{\mathrm{tr}}^{2}\bar{r}_{\perp}^{2}}{4}\frac{\delta}{\delta \phi_{a,r}}\sum_{c=1}^{2}\left( \phi_{c,r}\phi_{c,r-\hat{\tau}} + i \sum_{b=1}^{2}\epsilon_{cb}  \phi_{c,r}\phi_{b,r-\hat{\tau}}\right) \nonumber \\
& =&\frac{\bar{m}\bar{\omega}_{\mathrm{tr}}^{2}\bar{r}_{\perp}^{2}}{4}\left(\phi_{a,r-\hat{\tau}}+\phi_{a,r+\hat{\tau}} + i \sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r-\hat{\tau}}+\phi_{b,r+\hat{\tau}})\right) \nonumber 
\eea
%
where $\bar{r}_{\perp}^{2} = \bar{x}^{2} + \bar{y}^{2}$, or in $1d$, $\bar{r}_{\perp}^{2} = \bar{x}^{2}$. Next, the rotational piece:
%
\bea
\frac{\delta}{\delta \phi_{a,r}}  S_{\omega,r}& =& \frac{  \bar{\omega}_{z}}{2} \frac{\delta}{\delta \phi_{a,r}}\sum_{c=1}^{2} \left[\sum_{b=1}^{2}\epsilon_{cb}\left((\bar{y} - \bar{x})\phi_{c,r}\phi_{b,r-\hat{\tau}}+ \bar{x}\phi_{c,r}\phi_{b,r - \hat{y}-\hat{\tau}} - \bar{y}\phi_{c,r}\phi_{b,r - \hat{x}-\hat{\tau}}\right)\right.\nonumber\\  
&&  + i \Big((\bar{x} - \bar{y}) \phi_{c,r}\phi_{c,r-\hat{\tau}}- \bar{x} \phi_{c,r}\phi_{c,r - \hat{y}-\hat{\tau}} + \bar{y} \phi_{c,r}\phi_{c,r - \hat{x}-\hat{\tau}}\Big)\Bigg]\nonumber \\
& =& \frac{  \bar{\omega}_{z}}{2} \left[\sum_{b=1}^{2}\epsilon_{ab}\Big((\bar{y} - \bar{x})(\phi_{b,r-\hat{\tau}}+\phi_{b,r+\hat{\tau}})+ \bar{x}(\phi_{b,r - \hat{y}-\hat{\tau}} +\phi_{b,r + \hat{y}+\hat{\tau}} ) \right. \nonumber \\
&& \left. - \bar{y}(\phi_{b,r - \hat{x}-\hat{\tau}}+\phi_{b,r + \hat{x}+\hat{\tau}})\Big)  + i \Big((\bar{x} - \bar{y})(\phi_{a,r-\hat{\tau}}+\phi_{a,r+\hat{\tau}}) \right.\nonumber\\  
&&  - \bar{x} (\phi_{a,r - \hat{y}-\hat{\tau}} +\phi_{a,r + \hat{y}+\hat{\tau}})+ \bar{y}(\phi_{a,r - \hat{x}-\hat{\tau}}+\phi_{a,r + \hat{x}+\hat{\tau}})\Big)\Bigg]
%
\eea
%
Note, this term is set to zero unless $d >1$. And finally, the interaction term in the action:
%
\bea
\frac{\delta}{\delta \phi_{a,r}} S_{\text{int},r}& =&\frac{ \bar{\lambda}}{4}\frac{\delta}{\delta \phi_{a,r}}\sum_{c=1}^{2}\sum_{b=1}^{2}\left[2 \phi_{c,r} \phi_{c,r-\hat{\tau}} \phi_{b,r}\phi_{b,r-\hat{\tau}} -\phi_{c,r}^{2} \phi_{b,r-\hat{\tau}}^{2} \right. \nonumber \\
&& \left.+ 2 i \epsilon_{cb} \left( \phi_{c,r}^{2} \phi_{c,r-\hat{\tau}} \phi_{b,r-\hat{\tau}}-\phi_{c,r} \phi_{b,r}\phi_{c,r-\hat{\tau}}^{2} \right) \right] \nonumber \\
& =&\frac{ \bar{\lambda}}{4}\sum_{b=1}^{2}\left[ 4 \phi_{b,r}( \phi_{a,r+\hat{\tau}}\phi_{b,r+\hat{\tau}}+ \phi_{a,r-\hat{\tau}}\phi_{b,r-\hat{\tau}}) - 2 \phi_{a,r}(\phi_{b,r+\hat{\tau}}^{2} + \phi_{b,r-\hat{\tau}}^{2})\right.  \\
&& \left. + 2 i \epsilon_{ab}(\phi_{b,r}(\phi_{a,r+\hat{\tau}}^{2}+\phi_{a,r-\hat{\tau}}^{2} ) - \phi_{b,r}(\phi_{b,r+\hat{\tau}}^{2} + \phi_{b,r-\hat{\tau}}^{2})+ 2 \phi_{a,r}(\phi_{a,r-\hat{\tau}}\phi_{b,r-\hat{\tau}}  + \phi_{a,r+\hat{\tau}}\phi_{b,r+\hat{\tau}} ))\right]\nonumber
\eea
%
So our final drift function is given by:
%
\bea
K_{a} &=& \sum_{r} \Bigg[\phi_{a,r}- \frac{1}{2} e^{\bar{\mu}}\left(\phi_{a,r - \hat{\tau}}+\phi_{a,r + \hat{\tau}}\right) - \frac{i}{2} e^{\bar{\mu}}\epsilon_{ab}\left(\phi_{b,r - \hat{\tau}} +\phi_{b,r + \hat{\tau}}\right) \nonumber \\
&& + \frac{1}{2 \bar{m}}\sum_{j=1}^{d}\left[ 2\phi_{a,r} - (\phi_{a,r + \hat{j}}+\phi_{c,r - \hat{j}}) -  i\sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r + \hat{j}}+\phi_{b,r - \hat{j}}) \right] \nonumber \\
&& + \frac{\bar{m}\bar{\omega}_{\mathrm{tr}}^{2}\bar{r}_{\perp}^{2}}{4}\left(\phi_{a,r-\hat{\tau}}+\phi_{a,r+\hat{\tau}} + i \sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r-\hat{\tau}}+\phi_{b,r+\hat{\tau}})\right) \nonumber \\
&& +\frac{  \bar{\omega}_{z}}{2} \Bigg(\sum_{b=1}^{2}\epsilon_{ab}\Big((\bar{y} - \bar{x})(\phi_{b,r-\hat{\tau}}+\phi_{b,r+\hat{\tau}})+ \bar{x}(\phi_{b,r - \hat{y}-\hat{\tau}} +\phi_{b,r + \hat{y}+\hat{\tau}} )  \nonumber \\
&&  - \bar{y}(\phi_{b,r - \hat{x}-\hat{\tau}}+\phi_{b,r + \hat{x}+\hat{\tau}})\Big)  + i \Big((\bar{x} - \bar{y})(\phi_{a,r-\hat{\tau}}+\phi_{a,r+\hat{\tau}}) \nonumber\\  
&&  - \bar{x} (\phi_{a,r - \hat{y}-\hat{\tau}} +\phi_{a,r + \hat{y}+\hat{\tau}})+ \bar{y}(\phi_{a,r - \hat{x}-\hat{\tau}}+\phi_{a,r + \hat{x}+\hat{\tau}})\Big)\Bigg)\nonumber \\
&&+ \frac{ \bar{\lambda}}{4}\sum_{b=1}^{2}\left[ 4 \phi_{b,r}( \phi_{a,r+\hat{\tau}}\phi_{b,r+\hat{\tau}}+ \phi_{a,r-\hat{\tau}}\phi_{b,r-\hat{\tau}}) - 2 \phi_{a,r}(\phi_{b,r+\hat{\tau}}^{2} + \phi_{b,r-\hat{\tau}}^{2})\right.  \nonumber \\
&& \left. + 2 i \epsilon_{ab}(\phi_{b,r}(\phi_{a,r+\hat{\tau}}^{2}+\phi_{a,r-\hat{\tau}}^{2} ) - \phi_{b,r}(\phi_{b,r+\hat{\tau}}^{2} + \phi_{b,r-\hat{\tau}}^{2})+ 2 \phi_{a,r}(\phi_{a,r-\hat{\tau}}\phi_{b,r-\hat{\tau}}  + \phi_{a,r+\hat{\tau}}\phi_{b,r+\hat{\tau}} ))\right]
 \Bigg] \nonumber
\eea
%
Which we can divide into real and imaginary contributions: $K_{a} = K_{a}^{\mathrm{R}} + i K_{a}^{\mathrm{I}}$:
%
\bea 
K_{a}^{\mathrm{R}} &=& \sum_{r} \Bigg[
\phi_{a,r}- \frac{1}{2} e^{\bar{\mu}}\left(\phi_{a,r - \hat{\tau}}+\phi_{a,r + \hat{\tau}}\right)  
+ \frac{1}{2 \bar{m}}\sum_{j=1}^{d}\Big( 2\phi_{a,r} - (\phi_{a,r + \hat{j}}+\phi_{c,r - \hat{j}}) \Big) \nonumber \\
&& + \frac{\bar{m}\bar{\omega}_{\mathrm{tr}}^{2}\bar{r}_{\perp}^{2}}{4}\left(\phi_{a,r-\hat{\tau}}+\phi_{a,r+\hat{\tau}} \right)
+ \frac{  \bar{\omega}_{z}}{2} \sum_{b=1}^{2}\epsilon_{ab}\Bigg((\bar{y} - \bar{x})(\phi_{b,r-\hat{\tau}}+\phi_{b,r+\hat{\tau}}) \nonumber \\
&& + \bar{x}(\phi_{b,r - \hat{y}-\hat{\tau}} +\phi_{b,r + \hat{y}+\hat{\tau}} )  - \bar{y}(\phi_{b,r - \hat{x}-\hat{\tau}}+\phi_{b,r + \hat{x}+\hat{\tau}})\Bigg)  \nonumber \\
&&+ \frac{ \bar{\lambda}}{4}\sum_{b=1}^{2}\Big( 4 \phi_{b,r}( \phi_{a,r+\hat{\tau}}\phi_{b,r+\hat{\tau}}+ \phi_{a,r-\hat{\tau}}\phi_{b,r-\hat{\tau}}) - 2 \phi_{a,r}(\phi_{b,r+\hat{\tau}}^{2} + \phi_{b,r-\hat{\tau}}^{2})\Big)
 \Bigg] \nonumber \\
 %
 %
K_{a}^{\mathrm{I}} &=& - \sum_{r} \Bigg[\frac{1}{2} e^{\bar{\mu}}\epsilon_{ab}\left(\phi_{b,r - \hat{\tau}} +\phi_{b,r + \hat{\tau}}\right) 
+ \frac{1}{2 \bar{m}}\sum_{j=1}^{d}\Big(\sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r + \hat{j}}+\phi_{b,r - \hat{j}})  \Big) \nonumber \\
&& - \frac{\bar{m}\bar{\omega}_{\mathrm{tr}}^{2}\bar{r}_{\perp}^{2}}{4}\left( \sum_{b=1}^{2}\epsilon_{ab} (\phi_{b,r-\hat{\tau}}+\phi_{b,r+\hat{\tau}})\right)
- \frac{  \bar{\omega}_{z}}{2} \Bigg( \Big((\bar{x} - \bar{y})(\phi_{a,r-\hat{\tau}}+\phi_{a,r+\hat{\tau}}) \nonumber \\
&& - \bar{x} (\phi_{a,r - \hat{y}-\hat{\tau}} +\phi_{a,r + \hat{y}+\hat{\tau}}) + \bar{y}(\phi_{a,r - \hat{x}-\hat{\tau}}+\phi_{a,r + \hat{x}+\hat{\tau}})\Big)\Bigg) \nonumber \\
&& - \frac{ \bar{\lambda}}{2}\sum_{b=1}^{2}\Bigg( \epsilon_{ab}(\phi_{b,r}(\phi_{a,r+\hat{\tau}}^{2}+\phi_{a,r-\hat{\tau}}^{2} ) - \phi_{b,r}(\phi_{b,r+\hat{\tau}}^{2} + \phi_{b,r-\hat{\tau}}^{2})\nonumber \\
&& + 2 \phi_{a,r}(\phi_{a,r-\hat{\tau}}\phi_{b,r-\hat{\tau}}  + \phi_{a,r+\hat{\tau}}\phi_{b,r+\hat{\tau}} )) \Bigg)
\Bigg]
\eea
